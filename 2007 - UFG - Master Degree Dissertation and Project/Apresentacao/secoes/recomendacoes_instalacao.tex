\section{Recomendações para Instalação}

\subsection{Dispositivo Móvel}
\begin{frame}
		\frametitle{Recomendações para Instalação}
		\framesubtitle{Dispositivo Móvel}

		
\begin{itemize}
	\item Para instalar o Fênix Firewall System no Dispositivo Móvel é necessário que:
			\begin{itemize}
				\item O Dispositivo móvel possua \textbf{Sistema Operacional}.
				\item \textbf{Sistema Operacional} que possua recursos de filtragem de pacotes TCP/IP.
				\item \textbf{API} para utilizar os recursos de filtragem de pacotes.
				\item \textbf{API} para utilizar os recurso de Interface Gráfica com o Usuário.
				\item \textbf{Interface} de rede 802.11 compatível.
				\item \textbf{Memória} de armazenamento \textbf{disponível} maior ou igual a 20Mb.
				\item \textbf{Processador} com freqüência superior a 100Mhz.
			\end{itemize}
\end{itemize}

\end{frame}

\subsection{Plataforma Distribuída}
\begin{frame}
		\frametitle{Recomendações para Instalação}
		\framesubtitle{Plataforma Distribuída}

\begin{itemize}
	\item Para instalar a plataforma Distribuída do Fênix Firewall System \textbf{(Opcional)}:
			\begin{itemize}
				\item \textbf{Sistema Operacional} que suporte Java.
				\item \textbf{Interface de rede} compatível com protocolo TCP/IP.
				\item \textbf{Banco de Dados} relacional compatível com SQL.
				\item \textbf{Endereço IP} válido no domínio da Internet.
				\item \textbf{Domínio} válido registrado.				
				\item \textbf{Memória} secundária \textbf{disponível} maior ou igual a 100Mb.
				\item \textbf{Memória} principal (RAM) maior ou igual a 512Mb. 
				\item \textbf{Processador} com freqüência superior a 600Mhz.
			\end{itemize}
\end{itemize}

\end{frame}
